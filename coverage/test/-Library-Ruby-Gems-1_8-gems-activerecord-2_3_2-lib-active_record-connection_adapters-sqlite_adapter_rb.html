363         def copy_table_indexes(from, to, rename = {}) #:nodoc:
364           indexes(from).each do |index|
365             name = index.name
366             if to == "altered_#{from}"
367               name = "temp_#{name}"
368             elsif from == "altered_#{to}"
369               name = name[5..-1]
370             end

372             to_column_names = columns(to).map(&:name)
373             columns = index.columns.map {|c| rename[c] || c }.select do |column|
